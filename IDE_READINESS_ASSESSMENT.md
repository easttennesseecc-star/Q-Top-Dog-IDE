**STEP 1: Create UI Component** (30 min)
```jsx
// frontend/src/components/CodeAnalyzer.tsx
export function CodeAnalyzer() {
  const [code, setCode] = useState('');
  const [analysis, setAnalysis] = useState('');
  
  const handleAnalyze = async () => {
    const response = await fetch('/api/analyze', {
      method: 'POST',
      body: JSON.stringify({ code }),
    });
    setAnalysis(await response.json());
  };
  
  return (
    <div>
      <textarea value={code} onChange={e => setCode(e.target.value)} />
      <button onClick={handleAnalyze}>Analyze Code</button>
      <div>{analysis}</div>
    </div>
  );
}
```

**STEP 2: Create API Endpoint** (30 min)
```python
# backend/main.py (add to existing routes)
@app.post("/api/analyze")
async def analyze_code(request: dict):
    code = request.get("code")

        f"Analyze this code for issues:\n\n{code}"
    )
    
    return {"analysis": response}
```
